## Source

```jsx
import React from 'react';
import {
  Graph,
  compose,
  withViewportSize,
  withLayout,
  connectionLayout,
  chopBox,
} from '@regraph/graph';
import { Line } from '@regraph/connections';
import { TriangleMarker, PerpMarker } from '@regraph/arrowheads';

const MyGraph = compose(
  withLayout(connectionLayout(chopBox)),
  withViewportSize({ padding: ({ x, y }) => ({ right: x, bottom: y }) })
)(Graph);

export default () => (
  <MyGraph
    nodes={[{ id: 'ping' }, { id: 'pong' }]}
    boxes={{
      ping: { x: 50, y: 40, width: 20, height: 20 },
      pong: { x: 150, y: 40, width: 20, height: 20 },
    }}
    renderSvgNode={({ id, box }) => <rect key={id} {...box} fill="Chocolate" />}
    connections={[
      {
        id: 'ping->pong',
        src: 'ping',
        dst: 'pong',
      },
    ]}
    renderConnection={props => (
      <Line
        key={props.id}
        {...props}
        strokeWidth={2}
        markerStart={<PerpMarker width={6} height={6} />}
        markerEnd={<TriangleMarker width={6} height={6} />}
      />
    )}
  />
);
```

> ⚠️ The markers in this example are constants, so React will call `createElement` on every render:
>
> ```jsx
> markerEnd={<TriangleMarker width={6} height={6} />}
> ```
>
> To prevent this, either:
>
> - Use [@babel/plugin-transform-react-constant-elements](https://babeljs.io/docs/en/babel-plugin-transform-react-constant-elements) in production builds.
> - Define it as const outside the render function:
>
> ```jsx
> const markerEnd = <TriangleMarker width={6} height={6} />;
> ```
>
> Then:
>
> ```jsx
> markerEnd = { markerEnd };
> ```
